def Plot_decision_boundary(model,X,y):
    amin,bmin=X.min(axis=0)-0.1#subtracting -0.1 even from axis =0
    amax,bmax=X.max(axis=0)+0.1
    hticks=np.linspace(amin,amax,101)#we take the horizontal ticks(max x - min x)/101=3/101=0.03 se aage bad rhe points and vertical ticks
    vticks=np.linspace(bmin,bmax,101)
    aa,bb=np.meshgrid(hticks,vticks)#first layer having each point of hticks and vticks
    ab=np.c_[aa.ravel(),bb.ravel()]#ravel converts them into array and c_ converts both 1d array into 2d array of coordinates
    #?np.c_ for more details
    c=model.predict(ab)#on x and y coordinates 
    cc=c.reshape(aa.shape)
    plt.figure(figsize=(12,8))
    plt.contourf(aa,bb,cc,cmap='bwr',alpha=0.2)#this cmap is changing color from blue white red
    plt.plot(X[y==0,0],X[y==0,1],'ob',alpha=0.5)
    plt.plot(X[y==1,0],X[y==1,1],'xr',alpha=0.5)
    plt.legend(['Blue','Red'])
